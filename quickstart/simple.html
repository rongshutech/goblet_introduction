  • mainController.go
package controller

import (
	"github.com/extrame/goblet"
)

type MainController struct {
	goblet.RestController
	goblet.Route  "/index"
	goblet.Render"json,html"

}

func (mc *MainController) ReadMany(cx *goblet.Context) {
	cx.RespondOK()
}
  • main.go
package main

import (
	"github.com/extrame/goblet"
	"controller"
)

func main() {
	server := goblet.Organize("test")
	server.ControlBy(&controller.MainController{})
	server.Run()
}
